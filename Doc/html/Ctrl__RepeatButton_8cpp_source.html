00043 void CRepeatButton::OnLButtonDown(UINT nFlags, CPoint point) 
00044 {
00045     // Set up for notifying parent window
00046     int ControlID = GetDlgCtrlID();
00047 
00048     wParam = ControlID | (BN_CLICKED << 16);
00049     lParam = GetSafeHwnd();
00050 
00051     // Unlike an ordinary button, this sends the BN_CLICKED notification upon
00052     // Left Mouse Button DOWN rather than UP.
00053     NotifyParent(); // Send the click message to the parent for normal processing
00054 
00055     // initial timer interval = keyboard repeat delay
00056     int setting = 0;
00057     SystemParametersInfo(SPI_GETKEYBOARDDELAY, 0, &setting, 0);
00058     int InitialDelay = (setting + 1) * 250;
00059     TimerID = SetTimer(99, InitialDelay, NULL);
00060     TimerStep = 1;  // waiting for initial delay before starting auto-repeat
00061 
00062     // If you pass the message along to the base class then when you get a button up
00063     // you get an extra BN_CLICKED action. If you don't pass along the message
00064     // then the button state is not handled. So explicitly call SetState()
00065     // CButton::OnLButtonDown(nFlags, point);
00066     CButton::SetState(TRUE);
00067 }

00087 void CRepeatButton::OnTimer(UINT nIDEvent) 
00088 {
00089     // We may have moved the mouse during the timer interval so double-check it first!
00090     if(!IsOverClient()) {
00091         KillTimer(TimerID);
00092         TimerStep = 0;  // No longer doing auto-repeat
00093         CButton::SetState(FALSE);   // clear the depressed pushbutton appearence
00094         Invalidate(FALSE);  // stop with the yellow background already!
00095     }
00096     else {
00097         if(TimerStep == 1) {    // initial delay before auto-repeat?
00098             KillTimer(TimerID); // set up new timer interval
00099 
00100             // set timer interval per keyboard repeat rate
00101             int setting = 0;
00102             SystemParametersInfo(SPI_GETKEYBOARDSPEED, 0, &setting, 0);
00103             int RepeatInterval = 400 - (setting * 12);
00104             TimerID = SetTimer(100, RepeatInterval, NULL);
00105             TimerStep = 2;  // Now we are in auto-repeat mode
00106             Invalidate(FALSE);  // Change the background color to show repeat mode
00107         }
00108 
00109         if(TimerStep > 1) {
00110             NotifyParent(); // Send another click message to parent
00111         }
00112     }
00113 
00114     CButton::OnTimer(nIDEvent);
00115 }
