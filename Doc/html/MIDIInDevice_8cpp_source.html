00484 void CALLBACK CMIDIInDevice::MidiInProc(HMIDIIN MidiIn, UINT Msg,
00485                                         DWORD_PTR Instance, DWORD_PTR Param1,
00486                                         DWORD_PTR Param2)
00487 {
00488     CMIDIInDevice *Device;
00489     
00490     Device = reinterpret_cast<CMIDIInDevice *>(Instance);
00491 
00492     if(Device->m_Receiver != 0)
00493     {
00494         switch(Msg)
00495         {
00496         case MIM_DATA:      // Short message received
00497             Device->m_Receiver->ReceiveMsg(Param1, Param2);
00498             break;
00499 
00500         case MIM_ERROR:     // Invalid short message received
00501             Device->m_Receiver->OnError(Param1, Param2);
00502             break;
00503 
00504         case MIM_LONGDATA:  // System exclusive message received
00505             if(Device->m_State == RECORDING)
00506             {
00507                 // Retrieve data, send it to receiver, and notify header
00508                 // thread that we are done with the system exclusive 
00509                 // message
00510                 MIDIHDR *MidiHdr = reinterpret_cast<MIDIHDR *>(Param1);
00511               
00512                                 Device->m_Receiver->ReceiveMsg(MidiHdr->lpData,
00513                                                                                              MidiHdr->dwBytesRecorded, 
00514                                                Param2);
00515 
00516                 ::SetEvent(Device->m_Event);
00517             }
00518             break;
00519 
00520         case MIM_LONGERROR: // Invalid system exclusive message received
00521             if(Device->m_State == RECORDING)
00522             {
00523                 // Retrieve data, send it to receiver, and notify header
00524                 // thread that we are done with the system exclusive 
00525                 // message
00526                 MIDIHDR *MidiHdr = reinterpret_cast<MIDIHDR *>(Param1);
00527                 Device->m_Receiver->OnError(MidiHdr->lpData,
00528                                             MidiHdr->dwBytesRecorded,
00529                                             Param2);
00530                 ::SetEvent(Device->m_Event);
