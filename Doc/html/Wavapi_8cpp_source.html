00032 //          memcpy(lpDataOut, lpDataIn, DataSize);

00036             lpDataIn_16 = (short *) ::GlobalAlloc(GMEM_FIXED | GMEM_ZEROINIT, (DWORD)DataSize*sizeof(short));
00037             memcpy(lpDataOut_16, lpDataIn, DataSize);
00038             memcpy(lpDataIn_16, lpDataIn, DataSize);
00039             break;
00040     }
00041 
00042 //  DataWidth = wBitsPerSample >> 3;
00043     for( c = 0; c <= DataSize ; c += 2 ) 
00044     {
00045         TargetCount = c / 2 ;
00046         switch (wBitsPerSample)
00047         {
00048         case 8: lpDataOut[TargetCount] = ((unsigned char)lpDataIn[c] + (unsigned char)lpDataIn[c+1]) / 2;
00049                 break;
00050         case 16:
00051                 lpDataOut_16[TargetCount] = ((short)lpDataIn_16[c] + (short)lpDataIn_16[c + 1]) / 2;
00052                 break;

00054     }
00055     if ( wBitsPerSample == 8 )
00056     {
00057         memcpy(lpDataIn, lpDataOut, TargetCount);

00059     } else {
00060         memcpy(lpDataIn, lpDataOut_16, TargetCount);
