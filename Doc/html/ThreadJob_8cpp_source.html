00021  // Thread proc calls virtual DoWork function. This converts the
00022  // Windows/C-style thread procedure into an MFC/C++-style virtual function.
00023  // To do the "work" of the thread, implement DoWork and don't worry about
00024  // the thread proc.
00025  //
00026  UINT CThreadJob::ThreadProc(LPVOID pObj)
00027  {
00028      CThreadJob* pJob = (CThreadJob*)pObj;
00029      ASSERT_KINDOF(CThreadJob, pJob);
00030      pJob->m_uErr = pJob->DoWork();  // call virt fn to do the work
00031      pJob->m_pThread = NULL;         // done: clear
00032      return pJob->m_uErr;            // ..and return error code to Windows
00033  }

00036  // Begin running the worker thread. Args are owner window and callback
00037  // message ID to use for OnProgress notifications, if any. You could enhance
00038  // this to expose priority and other AfxBeginThread args.
00039  //
00040  BOOL CThreadJob::Begin(CWnd* pWndOwner, UINT ucbMsg)
00041  {
00042      m_hWndOwner = pWndOwner->GetSafeHwnd();
00043      m_ucbMsg = ucbMsg;
00044      m_bAbort = FALSE;
00045      m_uErr = 0;
00046      m_pThread = AfxBeginThread(ThreadProc, this);
00047      return m_pThread != NULL;
