00247 // Set note range
00248 BOOL CPianoCtrl::SetNoteRange(unsigned char LowNote, 
00249                               unsigned char HighNote)
00250 {
00251     BOOL Result = FALSE;
00252 
00253     // Range checking
00254     if(LowNote < 0)
00255     {
00256         MessageBox(LOW_NOTE_RANGE_ERR, "Error", MB_OK | MB_ICONSTOP);
00257     }
00258     // High note beyond maximum number allowed
00259     else if(HighNote >= MAX_NOTE_COUNT)
00260     {
00261         MessageBox(HIGH_NOTE_RANGE_ERR, "Error", MB_OK | MB_ICONSTOP);
00262     }
00263     // Low note is higher than high note
00264     else if(LowNote >= HighNote)
00265     {
00266         MessageBox(INVALID_RANGE_ERR, "Error", MB_OK | MB_ICONSTOP);
00267     }
00268     // Either the low note or the high note is not natural
00269     else if(!IsNoteNatural(LowNote) || !IsNoteNatural(HighNote))
00270     {
00271         MessageBox(NATURAL_NOTE_ERR, "Error", MB_OK | MB_ICONSTOP);
00272     }
00273     // Initialize range and create keys
00274     else
00275     {
00276         m_LowNote = LowNote;
00277         m_HighNote = HighNote;
00278 
00279         CreatePianoKeys();
00280 
00281         // Indicate that the operation was successful
00282         Result = TRUE;
00283     }
00284 
00285     return Result;
00286 }

00347     // The leftmost key is a special case. Take care of it first.
00348     //

00491 // Find a key in this control
00492 int CPianoCtrl::FindKey(CPoint &point)
00493 {
00494     bool FoundFlag = false;
00495     int Key = 0;
00496 
00497     // This algorithm does a simple linear search for the key that was
00498     // pressed. This is not an efficient way to search. However, it's
00499     // the simplest and works well enough in practise for this situation.
00500     while(!FoundFlag && Key < m_Keys.size() - 1)
00501     {
00502         if(m_Keys[Key]->IsPointInKey(point))
00503         {
00504             FoundFlag = true;
00505         }
00506         else
00507         {
00508             Key++;
00509         }
00510     }
00511 
00512     return Key;
00513 }

00608 // On left mouse button down
00609 void CPianoCtrl::OnLButtonDown(UINT nFlags, CPoint point) 
00610 {
00611     // Capture the mouse if we haven't already
00612     if(!m_HasCapture)
00613     {
00614         SetCapture();
00615         m_HasCapture = true;
00616     }
00617 
00618     // Make sure this control has been initialized
00619     if(m_IsInitialized)
00620     {
00621         // Find the key to turn on. This will become the current key. 
00622         // We keep track of it so that we can turn it off later
00623         m_CurrKey = FindKey(point);
00624 
00625         // Turn on note, get its area, and then invalidate so that the
00626         // control is repainted to indicate that this note is being 
00627         // played
00628         CRect InvalidRect;
00629         m_Keys[m_CurrKey]->NoteOn(m_NoteOnColor, InvalidRect);
00630 
00631         NotifyNoteOn(m_CurrKey + m_LowNote);    
00632 
00633         InvalidateRect(&InvalidRect);
00634     }
00635     
00636     CWnd::OnLButtonDown(nFlags, point);
00637 }
